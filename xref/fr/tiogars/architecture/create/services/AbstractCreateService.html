1   package fr.tiogars.architecture.create.services;
2   
3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   import org.springframework.data.repository.CrudRepository;
6   
7   import fr.tiogars.architecture.create.entities.AbstractEntity;
8   import fr.tiogars.architecture.create.forms.AbstractCreateForm;
9   import fr.tiogars.architecture.create.models.AbstractModel;
10  
11  /**
12   * Abstract class for create services.
13   * This class can be extended to implement specific create operations
14   * for different entities.
15   *
16   * @param <CreateForm> the type of the form used to create an entity
17   * @param <Entity> the type of the entity being created
18   * @param <Model> the type of the model used in the service
19   * @param <CreateRepository> the type of the repository used for CRUD operations
20   */
21  public abstract class AbstractCreateService<CreateForm extends AbstractCreateForm, Entity extends AbstractEntity, Model extends AbstractModel, CreateRepository extends CrudRepository<Entity, Long>> {
22  
23      /**
24       * The CrudRepository used for database operations.
25       * This repository should be injected in the constructor
26       * and is used to perform CRUD operations
27       * on the entity type.
28       */
29      public CrudRepository<Entity, Long> crudRepository;
30  
31      /**
32       * Logger for this service.
33       * Used to log information about operations performed by the service.
34       */
35      private final Logger logger = LoggerFactory.getLogger(AbstractCreateService.class);
36  
37      /**
38       * Default constructor for the AbstractCreateService.
39       * This constructor is provided for cases where dependency injection
40       * is not used.
41       */
42      public AbstractCreateService() {
43          // Empty constructor
44      }
45  
46      /**
47       * Constructor for the AbstractCreateService.
48       * Initializes the CrudRepository used for database operations.
49       * 
50       * @param crudRepository the CrudRepository for the entity
51       */
52      public AbstractCreateService(CreateRepository crudRepository) {
53          this.crudRepository = crudRepository;
54      }
55  
56      /**
57       * Converts a CreateForm to an Entity.
58       * This method should be implemented by subclasses to define how
59       * the conversion is done.
60       * 
61       * @param model the model containing data to create the entity
62       * @return the entity created from the model
63       */
64      public abstract Entity toEntity(Model model);
65  
66      /**
67       * Converts an UpdateRequest to an Entity.
68       * This method should be implemented by subclasses to define how
69       * the conversion is done.
70       * 
71       * @param entity the entity
72       * @return the model
73       */
74      public abstract Model toModel(Entity entity);
75  
76      /**
77       * Converts a CreateForm to a Model.
78       * This method should be implemented by subclasses to define how
79       * the conversion is done.
80       * 
81       * @param createForm the form containing data to create the model
82       * @return the model created from the form
83       */
84      public abstract Model toModel(CreateForm createForm);
85  
86      /**
87       * Validates the provided model.
88       * This method should be implemented by subclasses to define how
89       * the validation is done.
90       * 
91       * @param model the model to validate
92       */
93      public abstract void validate(Model model);
94  
95      /**
96       * Creates a new entity based on the provided CreateForm.
97       *
98       * @param createForm the form containing data to create the entity
99       * @return a response containing the ID of the created entity
100      */
101     public Model create(CreateForm createForm) {
102 
103         logger.info("Creating entity with data: {}", createForm);
104 
105         Model model = toModel(createForm);
106 
107         validate(model);
108 
109         Entity entity = toEntity(model);
110 
111         Entity savedEntity = crudRepository.save(entity);
112 
113         return toModel(savedEntity);
114     }
115 }
