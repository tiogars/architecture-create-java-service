179         TestCreateForm form = new TestCreateForm();
180         TestModel result = service.create(form);
181         assertTrue(toModelFromFormCalled, "toModel(createForm) should be called");
182         assertTrue(validateCalled, "validate(model) should be called");
183         assertTrue(toEntityCalled, "toEntity(model) should be called");
184         assertTrue(crudRepository.saveCalled, "crudRepository.save(entity) should be called");
185         assertTrue(toModelFromEntityCalled, "toModel(entity) should be called");
186         assertNotNull(result);
187     }
